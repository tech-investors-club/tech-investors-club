    <script>
        // Функция для прокрутки к секции
        function scrollToSection(sectionId) {
            document.getElementById(sectionId).scrollIntoView({
                behavior: 'smooth'
            });
        }

        // Обновление активной точки навигации при скролле
        document.addEventListener('DOMContentLoaded', function() {
            const sections = document.querySelectorAll('.section, .hero');
            const navDots = document.querySelectorAll('.nav-dot');
            const mobileNavItems = document.querySelectorAll('.mobile-nav-item');
            
            // Анимация появления элементов
            const fadeElements = document.querySelectorAll('.fade-in');
            
            const fadeInOnScroll = function() {
                fadeElements.forEach(element => {
                    const elementTop = element.getBoundingClientRect().top;
                    const elementVisible = 150;
                    
                    if (elementTop < window.innerHeight - elementVisible) {
                        element.style.opacity = "1";
                        element.style.transform = "translateY(0)";
                    }
                });
            };
            
            // Инициализация - скрываем элементы перед анимацией
            fadeElements.forEach(element => {
                element.style.opacity = "0";
                element.style.transform = "translateY(30px)";
                element.style.transition = "opacity 0.6s ease, transform 0.6s ease";
            });
            
            // Обновление навигации при скролле
            const updateNavigation = function() {
                let currentSection = '';
                
                sections.forEach(section => {
                    const sectionTop = section.offsetTop;
                    const sectionHeight = section.clientHeight;
                    if (pageYOffset >= (sectionTop - sectionHeight / 3)) {
                        currentSection = section.getAttribute('id');
                    }
                });
                
                // Обновление точек навигации
                navDots.forEach(dot => {
                    dot.classList.remove('active');
                    if (dot.getAttribute('data-title') === getSectionTitle(currentSection)) {
                        dot.classList.add('active');
                    }
                });
                
                // Обновление мобильной навигации
                mobileNavItems.forEach(item => {
                    item.classList.remove('active');
                    if (item.getAttribute('href') === '#' + currentSection) {
                        item.classList.add('active');
                    }
                });
            };
            
            // Вспомогательная функция для получения заголовка секции
            function getSectionTitle(sectionId) {
                const titles = {
                    'hero': 'Главная',
                    'features': 'Что строим',
                    'community': 'Основатель',
                    'roadmap': 'План',
                    'final-cta': 'Присоединиться'
                };
                return titles[sectionId] || '';
            }
            
            // Скрытие индикатора прокрутки после первого скролла
            let scrolled = false;
            const scrollIndicator = document.querySelector('.scroll-indicator');
            
            const hideScrollIndicator = function() {
                if (!scrolled) {
                    scrolled = true;
                    scrollIndicator.style.opacity = '0';
                    setTimeout(() => {
                        scrollIndicator.style.display = 'none';
                    }, 500);
                }
            };
            
            // Слушатели событий
            window.addEventListener('scroll', function() {
                updateNavigation();
                fadeInOnScroll();
                hideScrollIndicator();
            });
            
            // Инициализация при загрузке
            updateNavigation();
            fadeInOnScroll();
            
            // Обработчики для мобильной навигации
            mobileNavItems.forEach(item => {
                item.addEventListener('click', function(e) {
                    e.preventDefault();
                    const targetId = this.getAttribute('href').substring(1);
                    scrollToSection(targetId);
                });
            });
        });
    </script>
